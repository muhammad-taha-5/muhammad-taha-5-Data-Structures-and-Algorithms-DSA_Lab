#include<iostream>
#include<string>
using namespace std;

class node {
public:
    string name;
    node* next;
};

class patient_list {
private:
    node* head;
public:
    patient_list() {
        head = NULL;
    }

    void addPatient(string n) {
        node* newnode = new node;
        newnode->name = n;
        newnode->next = NULL;

        if (head == NULL) {
            head = newnode;
        } else {
            node* temp = head;
            while (temp->next != NULL) {
                temp = temp->next;
            }
            temp->next = newnode;
        }
    }

    void removePatient(string n) {
        if (head == NULL) return;

        if (head->name == n) {
            node* temp = head;
            head = head->next;
            delete temp;
            return;
        }

        node* pre = head;
        node* curr = head->next;

        while (curr != NULL) {
            if (curr->name == n) {
                pre->next = curr->next;
                delete curr;
                return;
            }
            pre = curr;
            curr = curr->next;
        }
    }

    void display() {
        node* temp = head;
        while (temp != NULL) {
            cout << temp->name << " ";
            temp = temp->next;
        }
        cout << endl;
    }

    int countPatients() {
        int c = 0;
        node* temp = head;
        while (temp != NULL) {
            c++;
            temp = temp->next;
        }
        return c;
    }
};

int main() {
    patient_list p;

    p.addPatient("Ali");
    p.addPatient("Ahmed");
    p.addPatient("Sara");

    p.display();
    cout << p.countPatients() << endl;

    p.removePatient("Ahmed");

    p.display();
    cout << p.countPatients() << endl;

    return 0;
}
